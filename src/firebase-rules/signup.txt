service cloud.firestore {
  match /databases/{database}/documents {
    
    // Rules for Voters collection
    match /Voters/{userId} {
      // Only authenticated users can create their own document. No one can read or write after creation.
      allow create: if request.auth.uid == userId;
      allow read, write: if false;
    }

    // Rules for User collection to store user roles
    match /User/{userId} {
      // Only authenticated users can create their own document. No one can read or write after creation.
      allow create: if request.auth.uid == userId;
      allow read, write: if false;
    }

    // General rules for other collections (if any)
    // ...
  }
}
